		mode : 'vertical',    
		auto : true, 
		autoHover : true, 
		pager : false, 
		controls : false, 
		pause : 3000,

		maxSlides: 4,

			slide1.startAuto();

			$('.exam_rolling li').removeClass('on');

	});

JS

<script type="text/javascript"> 
	var slider1;   //변수값 변경
	$(document).ready(function () {
		slider1 = $('#slider1').bxSlider({              //아이디/클래스명 변경
			//mode : 'fade',                           // 슬라이드 동작방법 
			speed : 500,                               // 슬라이드 이동속도
			pause : 3000,                             // 슬라이드 멈춰있는 시간
			controls : false,                          // 이전-다음 버튼  (직접 만들 시 false)
			pager : true,                              // 페이징 자동생성 (직접 만들시 pagerCustom 사용)
			//pagerCustom : '#btn_pager',                            // 페이징 직접생성&연결 : (div id="btn_pager") : 아이디/클래스명 변경가능
			auto : true,                               // 자동플레이
			autoHover : true,                       // 마우스 호버시 auto 정지 여부
			onSlideBefore : function (slide, oldIndex, newIndex){     
				slider1.startAuto();    
			}
		});
	});


	function go_slide1( bts ) {
		if ( bts == 0 ) { slider1.goToPrevSlide(); } //이전 버튼
		if ( bts == 1 ) { slider1.goToNextSlide(); } //다음 버튼
		if ( bts == 2 ) { slider1.stopAuto(); }  //슬라이드 멈춤 (사용 안할시 삭제)
		if ( bts == 3 ) { slider1.startAuto(); }  //슬라이드 동작 (사용 안할시 삭제)
	}
	
	//클릭시 auto 멈춤현상 해결 ( 이전다음/pager 버튼 클릭 후 auto 재실행하고 싶을때 )
	$(document).on('click','.bx-next, .bx-prev',function() {       //  .bx-next   .bx-prev -> 상황에 따라 클래스 변경
		slider1.stopAuto();                                                 //slider1  맞는지 확인 
		slider1.startAuto();
	});

</script>
