2.bxslider li 1개일때 bxslider 클래스삭제


//bxslider li 1개일때 bxslider 클래스삭제
$(document).ready(function(){
	$('.bxslider-default .bxslider').each(function(){
		if( $(this).find('li').length == 1 ){
			$(this).removeClass('bxslider');
		}
	});
});

<script>
	$(document).ready(function () {
		$('.js_tab_type li a').on('click', function (e) {
			var targetId = $(this).attr("href");
			if ($(this).attr("target") !== "_blank") {
				e.preventDefault();
				$(this).parent('li').siblings().find("a").removeClass('active');
				$(this).addClass('active');
				$(targetId).addClass('active').siblings().removeClass('active');
				if($(".bxslider-default").length > 0){
					__globalBxslider.reloadSlider();
				}
			} else {
				return true;
			}
		});
	});
</script>
