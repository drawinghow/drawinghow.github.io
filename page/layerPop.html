1. fixed 레이어팝업

HTML 예시

<a class="btn01" href="javascript:;" onclick="$('#ly_fixed').fadeIn();">fixed 레이어팝업</a>

<style>
	.layerPop {display: none; position: fixed; left: 0; top: 0; width: 100%; height: 100%; z-index: 100; text-align: center;}
	.layerPop .popInner { width: 550px; padding:50px; background: #fff;  margin: 150px auto 0; z-index: 999;  position: relative; text-align: left;}
	.layerPop .bt_popClose {  right: 25px; top: 25px; position: absolute; display: block; z-index: 1; text-align: left; text-}
	.layerPop .ly_dim { position: fixed; left: 0; top: 0; z-index: 990; width: 100%; height: 100%; background: rgba(0,0,0,.5); }
</style>

<!-- fixed 레이어팝업 -->
<div class="layerPop" id="ly_fixed">
	<div class="popInner">
		<!--  팝업내용  -->

		<a href="javascript:;" class="bt_popClose" onclick="$(this).parent().parent().fadeOut(100);">
			<img src="http://img.megaland.co.kr/icon/btn_close.png" alt="닫기"> <!-- 각사이트 마다 이미지 주소 바꿔줘야함-->
		</a>
	</div>
	<span class="ly_dim" onclick="$(this).parent().fadeOut(200);"></span>
</div>
<!-- //fixed 레이어팝업 -->

3. 이미지 레이어팝업

HTML 예시

<a href="javascript:;" class="btn01" onclick="layerPopupOpen('http://img.megamd.co.kr/2019/0919_kj/pop01.jpg')">이미지 레이어팝업</a>


<div id="layerPopImg" style="display: none; position: absolute; left: 0; top: 100px; z-index: 50; width: 100%; text-align: center;">
	<div style="position:relative;display: inline-block; z-index: 1;box-shadow: 26px 38px 26px 0 rgba(0, 0, 0, .3);">
		<img id="imgs" src="" alt=".">
		<a style="position: absolute; right: 0; top: 0; z-index: 2;" href="javascript:;" onclick="layerPopupOpen('no');">
			<img src="http://img.megaland.co.kr/icon/btn_close.png" alt="닫기"><!-- 각사이트 마다 이미지 주소 바꿔줘야함-->
		</a>
	</div>
</div>
<script type="text/javascript">
	function layerPopupOpen(imgurl){
	var position_top = $(this).scrollTop()-100;   //레이어 top 위치
	var imgsrc = imgurl

	$('#layerPopImg').css({ top : position_top });
	if( imgsrc == 'no' ){
		//레이어 닫기
		$('#layerPopImg').fadeOut(0);
		$('#imgs').attr('src', '');
	}else{
		//레이어 열기
		$('#layerPopImg').fadeIn(300);
		$('#imgs').attr('src', imgsrc);
	}
}
</script>
