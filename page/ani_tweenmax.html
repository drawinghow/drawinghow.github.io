CDN 링크

기본 문법

1. TweenMax.to( $('선택자'), 실행속도, { 실행요소 });
2. TweenMax.set( $('선택자'), { 실행요소 , onComplete:function(){
	//세팅 후 실행될 내용
}});
3. TweenMax.staggerTo( $('선택자들'), 실행속도, { 실행요소 }, 실행간격 );

실행 기능

width, height
backgroundColor
left, top
opacity,
delay,
scale, scaleX, scaleY
rotation, transformOrigin
repeat (repeat:-1 무한반복), repeatDelay, yoyo
ease: Power3.easeInOut,
onComplete: endFunction
